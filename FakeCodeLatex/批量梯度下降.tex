    \begin{algorithm}
        \caption{批量梯度下降}
        \begin{algorithmic}[1] %每行显示行号
            \Require $X$样本，$Y$样本, $\omega$初始化权重，$iter$迭代次数, $\alpha$学习率
            \Ensure $\omega$权重
            \Function {BatchGradientDesent}{$X, Y, \omega, iter, \alpha$}
                \State $m \gets X.shape[0]$ 
                \For {$i$ in $iter$}
                	\State $predict \gets X$ @ $\omega$
                	\State $grad \gets X.T$ @ $(predict - Y)$ / $m$ * $\alpha$
					\State $\omega \gets \omega$ - $grad$
                \EndFor
                \State \Return{$\omega$}
            \EndFunction
        \end{algorithmic}
    \end{algorithm}
\end{CJK*}
